glfw3.x86.dll	Windows x86 dynamic library

glfw3.x64.dll	Windows x64 dynamic library

glfw3.x86.lib	Windows x86 static library

glfw3.x64.lib	Windows x64 static library

glfw3.x64.so	Linux x64 dynamic library
